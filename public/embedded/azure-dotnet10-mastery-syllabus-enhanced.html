Regions, Availability Zones, Resource Hierarchy, IAM, Pricing

🔵 WHAT
Azure has 60+ regions, 200+ services, and 1000+ edge locations globally. Each region has multiple physically isolated Availability Zones.
🟢 WHY
Region choice affects latency, compliance, data residency, and availability SLA. AZs give higher resiliency than a single datacenter design.

Pick the region closest to users, deploy across zones for HA, and plan paired-region recovery for workloads that cannot afford a single-region dependency.

02

03
App Runtime: the request enters App Service, Functions, Container Apps, AKS, or VMs running your .NET code.

// Config from App Service Application Settings (Key Vault references)

// EF Core 10 with Azure SQL + transient-fault retry

    opts.UseSqlServer(builder.Configuration["SqlConnection"],
        sql => sql.EnableRetryOnFailure(maxRetryCount: 5)));

    .AddSqlServer(builder.Configuration["SqlConnection"]!, name: "sql")

// Required for App Service reverse proxy (X-Forwarded-* headers)
app.UseForwardedHeaders();

C# — Service Bus Trigger + Cosmos DB Output Binding
[Function("ProcessOrderMessage")]
public async Task ProcessOrder(
    [ServiceBusTrigger("orders-queue", Connection = "ServiceBusConnection")]
    ServiceBusReceivedMessage message,
    ServiceBusMessageActions messageActions,
    [CosmosDBOutput("orders-db", "processed", Connection = "CosmosConnection")]
    IAsyncCollector<OrderRecord> output,
    ILogger<OrderFunction> logger)
{
    try
    {
        var order = JsonSerializer.Deserialize<OrderPayload>(message.Body);
        logger.LogInformation("Processing order {OrderId}", order.Id);

        var processed = await _orderService.ProcessAsync(order);

        // Output binding — writes to Cosmos DB automatically (no SDK needed)
        await output.AddAsync(new OrderRecord { Id = order.Id, Status = "Processed" });
        await messageActions.CompleteMessageAsync(message);  // Acknowledge
    }
    catch (BusinessException ex)
    {
        await messageActions.DeadLetterMessageAsync(message, ex.Message);  // Poison message
    }
}

// Durable Function — Fan-Out/Fan-In parallel processing
[Function("InvoiceOrchestrator")]
public async Task<List<InvoiceResult>> RunOrchestrator(
    [OrchestrationTrigger] TaskOrchestrationContext context)
{
    var ids = await context.CallActivityAsync<List<int>>("GetPendingInvoices", null);

    // Fan-out: all invoices processed in parallel
    var tasks = ids.Select(id =>
        context.CallActivityAsync<InvoiceResult>("ProcessInvoice", id));

    var results = await Task.WhenAll(tasks);  // Fan-in: wait for all
    return results.ToList();
}

C# — Blob Storage: Upload, Download, SAS URL, Lifecycle
public class BlobStorageService(BlobServiceClient blobClient, IConfiguration config)
{
    private readonly string _container = config["Storage:Container"]!;

    // Upload file stream — returns blob URL
    public async Task<string> UploadAsync(string name, Stream content, string contentType)
    {
        var container = blobClient.GetBlobContainerClient(_container);
        await container.CreateIfNotExistsAsync(PublicAccessType.None);
        var blob = container.GetBlobClient(name);
        await blob.UploadAsync(content, new BlobUploadOptions
        {
            HttpHeaders = new BlobHttpHeaders { ContentType = contentType },
            Metadata = new Dictionary<string,string> { ["uploadedAt"] = DateTime.UtcNow.ToString("O") }
        });
        return blob.Uri.ToString();
    }

    // Pre-signed SAS URL — client uploads directly to blob (bypass API server)
    public string GenerateUploadSasUrl(string blobName, int expiryMinutes = 15)
    {
        var blob = blobClient.GetBlobContainerClient(_container).GetBlobClient(blobName);
        var sas = new BlobSasBuilder {
            BlobContainerName = _container, BlobName = blobName,
            Resource = "b",
            ExpiresOn = DateTimeOffset.UtcNow.AddMinutes(expiryMinutes)
        };

public class RedisCacheService(IConnectionMultiplexer redis)
{
    private IDatabase Db => redis.GetDatabase();

    // Cache-Aside: try cache → fallback to source → populate cache
    public async Task<T?> GetOrSetAsync<T>(string key, Func<Task<T?>> factory, TimeSpan? ttl = null)
    {
        var cached = await Db.StringGetAsync(key);
        if (cached.HasValue)
            return JsonSerializer.Deserialize<T>(cached!);

        var result = await factory();
        if (result != null)
            await Db.StringSetAsync(key, JsonSerializer.Serialize(result), ttl ?? TimeSpan.FromMinutes(5));
        return result;
    }

    // Sliding window rate limiter — Redis atomic INCR
    public async Task<bool> IsRateLimitedAsync(string clientId, int max, TimeSpan window)
    {
        var key = $"rl:{clientId}:{DateTime.UtcNow:yyyyMMddHHmm}";
        var count = await Db.StringIncrementAsync(key);
        if (count == 1) await Db.KeyExpireAsync(key, window);
        return count > max;
    }

    // Distributed lock — prevent duplicate processing across pods
    public async Task<bool> AcquireLockAsync(string resource, TimeSpan duration)
        => await Db.StringSetAsync($"lock:{resource}", "1", duration, When.NotExists);
}

// OrderConsumer.cs — IHostedService consumer (runs for lifetime of app)
public class OrderConsumer(ServiceBusClient bus, IOrderService svc) : BackgroundService
{
    protected override async Task ExecuteAsync(CancellationToken ct)
    {
        var processor = bus.CreateProcessor("orders-queue",
            new ServiceBusProcessorOptions { MaxConcurrentCalls = 10, AutoCompleteMessages = false });

        processor.ProcessMessageAsync += async args =>
        {
            var order = args.Message.Body.ToObjectFromJson<Order>();

            // Idempotency — skip already-processed messages (safe on retry)
            if (!await svc.IsAlreadyProcessedAsync(order.Id))
                await svc.ProcessAsync(order);

            await args.CompleteMessageAsync(args.Message);  // ACK
        };
        processor.ProcessErrorAsync += args => {
            /* log error, SDK retries automatically */
            return Task.CompletedTask;
        };

        await processor.StartProcessingAsync(ct);
        await Task.Delay(Timeout.Infinite, ct);
        await processor.StopProcessingAsync();
    }
}

public class RagService(AzureOpenAIClient client, SearchClient search)
{
    public async IAsyncEnumerable<string> AskAsync(string question)
    {
        // STEP 1: Embed the user question
        var embedClient = client.GetEmbeddingClient("text-embedding-3-large");
        var embed = await embedClient.GenerateEmbeddingAsync(question);

        // STEP 2: Hybrid vector + keyword search in AI Search
        var opts = new SearchOptions
        {
            VectorSearch = new VectorSearchOptions {
                Queries = { new VectorizedQuery(embed.Value.Vector)
                    { KNearestNeighborsCount = 5, Fields = { "ContentVector" } } }
            },
            Size = 5,
            Select = { "Content", "DocumentTitle" }
        };
        var results = await search.SearchAsync<DocChunk>(question, opts);
        var chunks = await results.Value.GetResultsAsync()
            .Select(r => r.Document.Content)
            .ToListAsync();

        // STEP 3: Build grounded prompt + stream GPT-4o response
        var context = string.Join("\n\n", chunks);
        var chatClient = client.GetChatClient("gpt-4o");
        var messages = new ChatMessage[] {
            new SystemChatMessage($"Answer ONLY from this context:\n{context}"),
            new UserChatMessage(question)
        };

        await foreach (var update in chatClient.CompleteChatStreamingAsync(messages))
            foreach (var t in update.ContentUpdate)
                yield return t.Text;
    }
}

Networking Cheat Sheet: decide three things first: your ingress type, your private connectivity type, and your egress control type. Most Azure network designs are combinations of those three choices.
Zero-Trust Network Reference
Use this as the default secure cloud posture: Front Door at the edge, private app ingress, managed identity, private endpoints for PaaS, and firewall-controlled outbound traffic.

If a service is production-critical and regulated, assume private access is the target until a public requirement proves otherwise.

Global HTTP Ingress

Use Front Door when traffic is internet-facing, global, cacheable, or needs WAF and failover at the edge.

  • Multi-region APIs
  • Static + dynamic acceleration

Regional App Ingress

Use Application Gateway when you need regional L7 routing, WAF, internal ingress, or private app exposure.

  • Internal enterprise apps
  • Private AKS or App Service patterns

Private PaaS Access

Use Private Endpoint or Private Link when SQL, Storage, Key Vault, or Service Bus must not expose public endpoints.

  • Regulated workloads
  • Zero-trust data access

Hybrid Connectivity

Use VPN for fast setup and ExpressRoute for predictable, private, enterprise-grade connectivity.

  • Dev/test: VPN
  • Mission-critical enterprise: ExpressRoute

02

name: Deploy to AKS
on: { push: { branches: [main] } }

permissions:
  id-token: write   # OIDC federation — no secrets in GitHub!
  contents: read

jobs:
  build-deploy:
    runs-on: ubuntu-latest
    steps:
    - uses: actions/checkout@v4

    - name: Setup .NET 10
      uses: actions/setup-dotnet@v4
      with: { dotnet-version: '10.0' }

    - name: Test
      run: |
        dotnet test --collect:"XPlat Code Coverage"
        dotnet publish -c Release -o ./publish

    - name: Login to Azure via OIDC
      uses: azure/login@v2
      with: { client-id: ${{ secrets.AZURE_CLIENT_ID }}, tenant-id: ${{ secrets.AZURE_TENANT_ID }}, subscription-id: ${{ secrets.AZURE_SUBSCRIPTION_ID }} }

    - name: Build + Push to ACR
      run: |
        az acr build --registry myacr --image order-service:${{ github.sha }} .

    - name: Deploy to AKS
      uses: azure/k8s-deploy@v4
      with:
        resource-group: my-rg
        name: my-aks
        manifests: k8s/
        images: myacr.azurecr.io/order-service:${{ github.sha }}

APIM Policy XML — JWT Validation + Rate Limit + Cache
<policies>
  <inbound>
    <!-- Validate JWT from Entra ID before hitting .NET backend -->
    <validate-jwt header-name="Authorization" failed-validation-httpcode="401">
      <openid-config url="https://login.microsoftonline.com/{tenantId}/.well-known/openid-configuration"/>
      <required-claims><claim name="aud"><value>api://my-dotnet-api</value></claim></required-claims>
    </validate-jwt>
    <!-- Rate limit: 100 calls per minute per subscription key -->
    <rate-limit-by-key calls="100" renewal-period="60" 
                       counter-key="@(context.Subscription.Id)"/>
    <!-- Cache GET responses for 30 seconds -->
    <cache-lookup vary-by-developer="false" vary-by-developer-groups="false"/>
    <!-- Forward user ID to .NET backend as header -->
    <set-header name="X-User-Id" exists-action="override">
      <value>@(context.Request.Headers["Authorization"].AsJwt()?.Subject)</value>
    </set-header>
  </inbound>
  <outbound>
    <cache-store duration="30"/>  <!-- cache 30s for GET -->
    <!-- Remove internal headers before returning to client -->
    <set-header name="X-Powered-By" exists-action="delete"/>
    <set-header name="Server" exists-action="delete"/>
  </outbound>
</policies>
