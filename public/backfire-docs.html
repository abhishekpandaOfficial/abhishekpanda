namespace BackFire.Core.Models;

public class Job
{
    public Guid      Id          { get; set; }
    public string    Type        { get; set; } = "";

    public string    Arguments   { get; set; } = "";

    public DateTime? ScheduledAt { get; set; }

    public string    Status      { get; set; } = "Queued";
}

Storage Abstraction Layer

6.1 Storage Interface

C# · IJobStorage 
public interface IJobStorage
{
    Task                    EnqueueAsync(Job job);
    Task<Job?>             FetchNextJobAsync(string queue);
    Task                    MarkProcessingAsync(Guid id);
    Task                    MarkCompletedAsync(Guid id);
    Task                    MarkFailedAsync(Guid id);

}
This interface allows supporting: SQL Server, PostgreSQL, Redis, MongoDB — and any future storage engine through the plugin system.

13

Job Executor

Responsible for executing job methods.

C# · JobExecutor
public async Task ExecuteAsync(Job job)
{
    var type     = Type.GetType(job.Type);
    var instance = Activator.CreateInstance(type!);
    var method   = type!.GetMethod(job.Method);
    var args     = JsonSerializer.Deserialize<object[]>(job.Arguments);

    var result = method!.Invoke(instance, args);

    if (result is Task task)
        await task;
}

Retry Policy System

Retries improve reliability.

AttemptStrategy
1st failureRetry immediately
2nd failureRetry after 10s
3rd failureRetry after 1m

if (job.RetryCount < 3)
{
    job.RetryCount++;
    await storage.EnqueueAsync(job);
}
else
{
    await storage.MarkFailedAsync(job.Id);
}
