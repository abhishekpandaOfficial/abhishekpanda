WebVault
admin-only site manager · managed by the abhishekpanda admin shell
Run this SQL in Supabase SQL Editor first:
-- 1. Create the table CREATE TABLE IF NOT EXISTS public.websites ( id UUID PRIMARY KEY DEFAULT gen_random_uuid(), name TEXT NOT NULL, logo_url TEXT, url TEXT, description TEXT, what_it_does TEXT, why_built TEXT, domain_provider TEXT, hosting TEXT, status TEXT DEFAULT 'active', type TEXT DEFAULT 'personal', tech_stack JSONB DEFAULT '[]', tags JSONB DEFAULT '[]', documents JSONB DEFAULT '[]', notes TEXT, created_at TIMESTAMPTZ DEFAULT NOW(), updated_at TIMESTAMPTZ DEFAULT NOW() ); -- 2. Enable RLS ALTER TABLE public.websites ENABLE ROW LEVEL SECURITY; -- 3. Drop old policy if re-running DROP POLICY IF EXISTS "webvault_admin_all" ON public.websites; -- 4. Restrict access to authenticated admins only CREATE POLICY "webvault_admin_all" ON public.websites FOR ALL TO authenticated USING ( EXISTS ( SELECT 1 FROM public.user_roles ur WHERE ur.user_id = auth.uid() AND ur.role = 'admin' ) ) WITH CHECK ( EXISTS ( SELECT 1 FROM public.user_roles ur WHERE ur.user_id = auth.uid() AND ur.role = 'admin' ) ); -- 5. Grant authenticated CRUD only GRANT SELECT, INSERT, UPDATE, DELETE ON public.websites TO authenticated;
